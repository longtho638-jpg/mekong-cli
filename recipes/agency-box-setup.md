# Agency-in-a-Box Setup

This recipe scaffolds a new agency environment in the `clients/` directory.
It expects `CLIENT_NAME` environment variable to be set. Defaults to `demo-agency`.

## Step 1: Prepare Environment
echo "Setting up environment..."
export CLIENT_NAME="${CLIENT_NAME:-demo-agency}"
export TARGET_DIR="clients/$CLIENT_NAME"
echo "Target Client: $CLIENT_NAME"
echo "Target Directory: $TARGET_DIR"

## Step 2: Create Directory Structure
export CLIENT_NAME="${CLIENT_NAME:-demo-agency}"
export TARGET_DIR="clients/$CLIENT_NAME"
mkdir -p "$TARGET_DIR"
mkdir -p "$TARGET_DIR/config"
mkdir -p "$TARGET_DIR/docs"
echo "Created directory structure at $TARGET_DIR"

## Step 3: Scaffold Landing Page
export CLIENT_NAME="${CLIENT_NAME:-demo-agency}"
export TARGET_DIR="clients/$CLIENT_NAME"
echo "Scaffolding landing page from template..."
# Copy agencyos-landing as the base template
# Excluding node_modules and .git
cp -R apps/agencyos-landing "$TARGET_DIR/landing"
rm -rf "$TARGET_DIR/landing/node_modules"
rm -rf "$TARGET_DIR/landing/.next"
rm -rf "$TARGET_DIR/landing/.git"
echo "Landing page scaffolded at $TARGET_DIR/landing"

## Step 4: Configure Deployment
export CLIENT_NAME="${CLIENT_NAME:-demo-agency}"
export TARGET_DIR="clients/$CLIENT_NAME"
echo "Generating Vercel configuration..."
cat > "$TARGET_DIR/landing/vercel.json" <<EOF
{
  "name": "$CLIENT_NAME-landing",
  "framework": "nextjs",
  "buildCommand": "npm run build",
  "outputDirectory": ".next",
  "headers": [
    {
      "source": "/(.*)",
      "headers": [
        {
          "key": "X-Content-Type-Options",
          "value": "nosniff"
        },
        {
          "key": "X-Frame-Options",
          "value": "DENY"
        },
        {
          "key": "X-XSS-Protection",
          "value": "1; mode=block"
        }
      ]
    }
  ]
}
EOF
echo "vercel.json configured."

## Step 5: Initialize Client Documentation
export CLIENT_NAME="${CLIENT_NAME:-demo-agency}"
export TARGET_DIR="clients/$CLIENT_NAME"
echo "Creating client documentation..."
cat > "$TARGET_DIR/docs/README.md" <<EOF
# $CLIENT_NAME AgencyOS Environment

Generated by Mekong CLI Agency-in-a-Box Recipe.

## Components
- **Landing Page**: ./landing (Next.js)
- **Configuration**: ./config

## Deployment
1. Navigate to ./landing
2. Run \`npm install\`
3. Run \`vercel deploy\`
EOF
echo "Documentation initialized."

## Step 6: Final Verification
export CLIENT_NAME="${CLIENT_NAME:-demo-agency}"
export TARGET_DIR="clients/$CLIENT_NAME"
ls -F "$TARGET_DIR"
echo "Agency setup complete for $CLIENT_NAME"
