- Implementing autonomous procurement and reorder systems
- Creating supply chain visibility and risk monitoring
- Optimizing logistics routing and warehouse operations

## Implementation Patterns
```typescript
interface SupplyChainConfig {
  forecastModel: 'prophet' | 'deepar' | 'arima' | 'ensemble';
  planningHorizon: number; // days
  reorderStrategy: 'fixed-point' | 'periodic' | 'demand-driven';
  safetyStockMethod: 'statistical' | 'ml-based';
}

interface DemandForecast {
  sku: string;
  period: string; // ISO date
  predicted: number;
  lowerBound: number;
  upperBound: number;
  confidence: number;
  drivers: { factor: string; impact: number }[];
}

interface PurchaseOrder {
  supplierId: string;
  items: { sku: string; quantity: number; unitCost: number }[];
  status: 'draft' | 'submitted' | 'confirmed' | 'received' | 'closed';
  expectedDelivery: string;
  autoGenerated: boolean;
}
```
